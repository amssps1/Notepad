use master
go
IF EXISTS (SELECT * FROM sys.server_principals WHERE name = 'COFIDIS2000\GRP_SQL_RH_TECNICO')
     DROP LOGIN [COFIDIS2000\GRP_SQL_RH_TECNICO];
GO  

IF NOT EXISTS (SELECT * FROM sys.server_principals WHERE name = 'COFIDIS2000\GRP_SQL_RH_TECNICO')
     CREATE LOGIN [COFIDIS2000\GRP_SQL_RH_TECNICO] FROM WINDOWS WITH DEFAULT_DATABASE=[master], DEFAULT_LANGUAGE=[us_english];
GO    


EXEC sp_MSforeachdb '
IF ''?'' IN (''RHInternalApp'') 
BEGIN 
    USE [?]
    PRINT ''?''
	DROP USER IF EXISTS [COFIDIS2000\GRP_SQL_RH_TECNICO]

    IF DATABASE_PRINCIPAL_ID(''COFIDIS2000\GRP_SQL_RH_TECNICO'') IS NULL
    BEGIN
        PRINT '' Criando o user.''
        CREATE USER [COFIDIS2000\GRP_SQL_RH_TECNICO] FROM LOGIN [COFIDIS2000\GRP_SQL_RH_TECNICO];
    END /*IF*/
    PRINT '' Adicionar as permissões para o user ...''
	GRANT CONNECT TO [COFIDIS2000\GRP_SQL_RH_TECNICO]
	GRANT SELECT  TO [COFIDIS2000\GRP_SQL_RH_TECNICO] 
	GRANT VIEW Definition 		TO [COFIDIS2000\GRP_SQL_RH_TECNICO]
	GRANT SHOWPLAN 				TO [COFIDIS2000\GRP_SQL_RH_TECNICO]

END /*IF*/
'


/*
--Privilégios MSDB
*/
USE [msdb]
GO
DROP SCHEMA IF EXISTS [COFIDIS2000\GRP_SQL_RH_TECNICO]
DROP USER IF EXISTS [COFIDIS2000\GRP_SQL_RH_TECNICO]
go
IF NOT EXISTS (SELECT NAME FROM sys.database_principals WHERE NAME =  'COFIDIS2000\GRP_SQL_RH_TECNICO') BEGIN CREATE USER  [COFIDIS2000\GRP_SQL_RH_TECNICO] 
   FOR LOGIN [COFIDIS2000\GRP_SQL_RH_TECNICO] END ELSE ALTER USER  [COFIDIS2000\GRP_SQL_RH_TECNICO] WITH LOGIN = [COFIDIS2000\GRP_SQL_RH_TECNICO];
IF DATABASE_PRINCIPAL_ID('COFIDIS2000\GRP_SQL_RH_TECNICO') IS NOT NULL GRANT CONNECT TO [COFIDIS2000\GRP_SQL_RH_TECNICO]

exec sp_addrolemember 'SQLAgentUserRole', 'COFIDIS2000\GRP_SQL_RH_TECNICO'
exec sp_addrolemember 'db_ssisoperator', 'COFIDIS2000\GRP_SQL_RH_TECNICO'
exec sp_addrolemember 'db_datareader', 'COFIDIS2000\GRP_SQL_RH_TECNICO'

/*
--Privilégios SSISDB
*/
USE [SSISDB]
GO
DROP USER IF EXISTS [COFIDIS2000\GRP_SQL_RH_TECNICO]
go
IF NOT EXISTS (SELECT NAME FROM sys.database_principals WHERE NAME =  'COFIDIS2000\GRP_SQL_RH_TECNICO') BEGIN CREATE USER  [COFIDIS2000\GRP_SQL_RH_TECNICO] 
   FOR LOGIN [COFIDIS2000\GRP_SQL_RH_TECNICO] END ELSE ALTER USER  [COFIDIS2000\GRP_SQL_RH_TECNICO] WITH LOGIN = [COFIDIS2000\GRP_SQL_RH_TECNICO];
IF DATABASE_PRINCIPAL_ID('COFIDIS2000\GRP_SQL_RH_TECNICO') IS NOT NULL GRANT CONNECT TO [COFIDIS2000\GRP_SQL_RH_TECNICO]

ALTER ROLE [db_datareader] ADD MEMBER [COFIDIS2000\GRP_SQL_RH_TECNICO]
GO
